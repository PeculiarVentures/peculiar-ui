## Overview

React hook that tracks state of a CSS media query.

## Usage

```jsx
import { useMediaQuery } from '@peculiar/react-components';

const Demo = () => {
  const isWide = useMediaQuery('(min-width: 480px)');

  return (
    <div>
      Screen is wide:
      {' '}
      {isWide ? 'Yes' : 'No'}
    </div>
  );
};
```

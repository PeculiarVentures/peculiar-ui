## Overview

React hook that provides copy to clipboard functionality.

## Usage

```jsx
import { useClipboard } from '@peculiar/react-components';

export const Demo = () => {
  const {
    onCopy,
    hasCopied,
  } = useClipboard('Text to copy');

  return (
    <button
      type="button"
      onClick={onCopy}
    >
      {hasCopied ? 'Copied' : 'Click to copy'}
    </button>
  );
};
```
